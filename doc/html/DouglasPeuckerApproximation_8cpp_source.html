87 void DouglasPeuckerApproximation::douglasPeucker(const std::vector <FrameWithId>& pointList,
88  std::vector <FrameWithId>& resultList, int& numberOfIterations, double epsilon) {
89 
90  ++numberOfIterations;
91  if (pointList.size() < 2) {
92  resultList = pointList;
93  return;
94  }
95 
96  double dmax = 0;
97  unsigned int index = 0;
98 
99  std::vector <FrameWithId> result1;
100  std::vector <FrameWithId> result2;
101 
102  for (unsigned int i = 1; i < pointList.size() - 1; i++) {
103  double d = utilities::perpendicularDistance(pointList[i], pointList[0],
104  pointList[pointList.size() - 1]);
105  if (d > dmax) {
106  index = i;
107  dmax = d;
108  }
109  }
110  //If max distance is greater than epsilon, recursively simplify
111  if (dmax >= epsilon) {
112  //Recursive call
113  std::vector <FrameWithId> pointSubList1(&pointList[0],
114  &pointList[index]); // FixMe: THis doesn't make sense, probably this should use the respective iterators.
115  std::vector <FrameWithId> pointSubList2(&pointList[index],
116  &pointList[pointList.size()]);
117 
118  douglasPeucker(pointSubList1, result1, numberOfIterations, epsilon);
119  douglasPeucker(pointSubList2, result2, numberOfIterations, epsilon);
120  resultList.insert(resultList.begin(), result1.begin(), result1.end() - 1);
121  resultList.insert(resultList.end(), result2.begin(), result2.end());
122  } else {
123  resultList.push_back(pointList[0]);
124  resultList.push_back(pointList[pointList.size() - 1]);
125  }
126 }
