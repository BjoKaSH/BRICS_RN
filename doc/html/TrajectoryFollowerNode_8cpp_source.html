80 void pose2dToPoseRos(const Pose2D& pose2d, geometry_msgs::Pose& pose) {

84  tf::Quaternion quat;

117  std::vector<FrameWithId> pathKDL;

121  KDL::Trajectory_Composite trajectoryKDL;

128 void publishOdometry(const Odometry& odometry) {
129 
130  geometry_msgs::Twist twist;
131  geometry_msgs::Pose pose;
132  pose2dToPoseRos(odometry.getPose2D(), pose);
133  twist2dToTwistRos(odometry.getTwist2D(), twist);
134 
135  nav_msgs::Odometry odomRos;
136  odomRos.pose.pose = pose;
137  odomRos.twist.twist = twist;
138  odomRos.header.stamp = ros::Time::now();
139 
140  twistPublisher.publish(twist);
141 
142 }
