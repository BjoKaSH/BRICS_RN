116 double controlLoop() {
117 
118  double defaultCycleFrequencyInHz = 10.0;
119 
120  switch (actualState.get()) {
121  case TrajectoryAdapterNodeState::IDLING:
122  {
123  // ROS_INFO("State IDLING");
124  break;
125  }
126  case TrajectoryAdapterNodeState::PLANNING:
127  {
128  ROS_INFO("State PLANNING");
129 
130  path.clear();
131  simplifiedPath.clear();
132 
133  DouglasPeuckerApproximation peucker;
134  ChaikinCurveApproximation chaikin;
135 
136  pathPlanner->computePath(actualPose, goalPose, path);
137 
138  if (path.empty())
139  ROS_WARN("Can't plan a path");
140 
141  peucker.approximate(path, simplifiedPath);
142 
143  // chaikin.approximate(simplifiedPath, simplifiedPath, 2);
144 
145  nav_msgs::Path pathRos, simplifiedPathRos;
146  conversions::pathToPathRos(path, pathRos);
147  conversions::pathToPathRos(simplifiedPath, simplifiedPathRos);
148 #ifdef DEBUG
149  pathPublisher.publish(pathRos);
150  simplifiedPathPublisher.publish(simplifiedPathRos);
151 #endif
152  actualState.set(TrajectoryAdapterNodeState::COLLISION_CHECKING);
153 
154  break;
155  }
156  case TrajectoryAdapterNodeState::COLLISION_CHECKING:
157  {
158  bool collision = collisionChecker->check(simplifiedPath, actualPose);
159  if (collision)
160  actualState.set(TrajectoryAdapterNodeState::PLANNING);
161  break;
162  }
163  default:
164  ROS_ERROR("Unknown state");
165  }
166 
167  return defaultCycleFrequencyInHz;
168 }
