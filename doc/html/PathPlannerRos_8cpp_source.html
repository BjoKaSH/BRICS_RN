61 bool PathPlannerRos::computePath(const FrameWithId& start, const FrameWithId& goal, std::vector <FrameWithId>& path) {
62 
63  if (!planner) {
64  ROS_ERROR("Planner instance in not allocated");
65  return false;
66  }
67 
68  geometry_msgs::PoseStamped startRos, goalRos;
69  conversions::frameToPoseStampedRos(start, startRos);
70  conversions::frameToPoseStampedRos(goal, goalRos);
71  std::vector<geometry_msgs::PoseStamped> pathRos;
72 
73 #ifdef DEBUG
74  Stopwatch stopwatch;
75  stopwatch.start();
76 #endif
77 
78  bool result = planner->makePlan(startRos, goalRos, pathRos);
79  if (result) {
80  pathRos.front() = startRos;
81  pathRos.back() = goalRos;
82  }
83 
84 #ifdef DEBUG
85  stopwatch.stop();
86  LOG("A* path planning:");
87  LOG(" - start: %f,%f,%f", startRos.pose.position.x, startRos.pose.position.y, tf::getYaw(startRos.pose.orientation));
88  LOG(" - goal: %f,%f,%f", goalRos.pose.position.x, goalRos.pose.position.y, tf::getYaw(goalRos.pose.orientation));
89  LOG(" - output size: %lu", pathRos.size());
90  LOG(" - duration : %f ms", stopwatch.getElapsedTime());
91 #endif
92 
93  conversions::pathRosToPath(pathRos, path);
94  return result;
95 }
