388  inline void pathRosToPathKdl(const nav_msgs::Path& pathRos, double eqradius, KDL::Path_Composite& path) {
389 
390  if (pathRos.poses.size() > 1) {
391 
392  std::vector<geometry_msgs::PoseStamped>::const_iterator it = pathRos.poses.begin();
393  geometry_msgs::PoseStamped p1 = *it;
394  it++;
395  geometry_msgs::PoseStamped p2;
396 
397  KDL::Frame f1Kdl;
398  poseRosToFrameKdl(p1.pose, f1Kdl);
399 
400  KDL::Frame f2Kdl;

402  p2 = *it;

405  KDL::Path_Line* pathLine = new KDL::Path_Line(f1Kdl, f2Kdl, new KDL::RotationalInterpolation_SingleAxis(), eqradius);
406  path.Add(pathLine, true);
407 
408  p1 = p2;
409  f1Kdl = f2Kdl;
410  }
411 
412  } else if (pathRos.poses.size() == 1) {
413  ROS_WARN("Adding a single point to KDL::Path_composite. Expect problems later on.");
414  std::vector<geometry_msgs::PoseStamped>::const_iterator it = pathRos.poses.begin();
415  geometry_msgs::PoseStamped p1 = *it;
416  KDL::Frame f1Kdl;
417  path.Add(new KDL::Path_Point(f1Kdl), true);

487  inline void rosKdlStreamToTrajectoryKdl(const navigation_trajectory_common::KdlStream &trajectoryMsg, KDL::Trajectory_Composite &trajectory) {
488 
489  std::stringstream buffer;
490  buffer.str(trajectoryMsg.KDLstream);
491  char tmpBuf[80];
492  KDL::EatWord(buffer, "[", tmpBuf, 80);
493  if (strcmp(tmpBuf, "COMPOSITE") != 0) {
494  ROS_WARN("Failed to read trajectory message!");
495  return;
496  }
497  KDL::Eat(buffer, "[");
498  int c = buffer.peek();
499  while (c != ']' && !buffer.eof()) {
500  trajectory.Add(KDL::Trajectory::Read(buffer));
501  c=buffer.peek();
502  }
503  }
