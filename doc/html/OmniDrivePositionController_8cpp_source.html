135 void OmniDrivePositionController::targetReached(bool& translation, bool& rotation) {
136 
137  KDL::Trajectory& trajectoryComposite = targetTrajectory.getTrajectory();
138 
139  double duration = trajectoryComposite.Duration();
140  KDL::Frame frame = trajectoryComposite.Pos(duration);
141 
142  double roll, pitch, yaw;
143  frame.M.GetRPY(roll, pitch, yaw);
144 
145  Pose2D desiredPose(frame.p.x(), frame.p.y(), yaw);
146  Pose2D actualPose = actualOdometry.getPose2D();
147 
148  double angDist = getShortestAngle(desiredPose.getTheta(), actualPose.getTheta());
149  double linDist = getDistance(desiredPose, actualPose);
150 
151  translation = false;
152  rotation = false;
153 
154  if (linDist <= tolerance.getPose2D().getX() && linDist <= tolerance.getPose2D().getY()) {
155  translation = true;
156  }
157 
158  if (fabs(angDist) <= tolerance.getPose2D().getTheta()) {
159  rotation = true;
160  }
161 }

163 const Odometry& OmniDrivePositionController::computeNewOdometry(const Odometry& actualOdometry, double elapsedTimeInSec) {
164 
165 
166  KDL::Trajectory& trajectoryComposite = targetTrajectory.getTrajectory();
167 
168  this->actualOdometry = actualOdometry;
169  computedOdometry = Odometry();
170 
171  if (trajectoryComposite.Duration() > 0 && !isTargetReached()) {
172 
173  actualTime = elapsedTimeInSec;
174 
175  KDL::Frame desiredPose = trajectoryComposite.Pos(actualTime);
176  KDL::Twist desiredTwist = trajectoryComposite.Vel(actualTime);
177  KDL::Frame initPose = trajectoryComposite.Pos(0);
178 
179  double r, p, y;
180  desiredPose.M.GetRPY(r, p, y);
181 
182  Odometry desiredOdometryGlobal(Pose2D(desiredPose.p.x(),
183  desiredPose.p.y(),
184  actualOdometry.getPose2D().getTheta()),
185  Twist2D(desiredTwist.vel.x(),
186  desiredTwist.vel.y(),
187  desiredTwist.rot.z()));
188 
189  Odometry desiredOdometryLocal;
190 
191  Odometry actualOdometryGlobal(actualOdometry);
192  Odometry actualOdometryLocal;
193 
194  OmniDriveKinematicsModel omnidrive;
195  omnidrive.convertToLocalReferenceFrame(desiredOdometryGlobal, desiredOdometryLocal);
196  omnidrive.convertToLocalReferenceFrame(actualOdometryGlobal, actualOdometryLocal);
197 
198  double dPosX = desiredOdometryLocal.getPose2D().getX();
199  double dPosY = desiredOdometryLocal.getPose2D().getY();
200  double dPosTheta = y; //desiredOdometryLocal.getPose2D().getTheta();
201 
202  double dVelX = desiredOdometryLocal.getTwist2D().getX();
203  double dVelY = desiredOdometryLocal.getTwist2D().getY();
204  double dVelTheta = desiredOdometryLocal.getTwist2D().getTheta();
205 
206  double aPosX = actualOdometryLocal.getPose2D().getX();
207  double aPosY = actualOdometryLocal.getPose2D().getY();
208  double aPosTheta = actualOdometryLocal.getPose2D().getTheta();
209 
210  double positionXError = dPosX - aPosX;
211  double positionYError = dPosY - aPosY;
212  double positionThetaError = getShortestAngle(dPosTheta, aPosTheta);
213 
214  double gainPosX = gains.getPose2D().getX();
215  double gainPosY = gains.getPose2D().getY();
216  double gainPosTheta = gains.getPose2D().getTheta();
217 
218  double gainVelX = gains.getTwist2D().getX();
219  double gainVelY = gains.getTwist2D().getY();
220  double gainVelTheta = gains.getTwist2D().getTheta();
221 
222  double errorTheta = gainVelTheta * dVelTheta + gainPosTheta * positionThetaError;
223  double errorX = gainVelX * dVelX + gainPosX * positionXError;
224  double errorY = gainVelY * dVelY + gainPosY * positionYError;
225 
226  computedOdometry.setTwist2D(Twist2D(errorX, errorY, errorTheta));
227  }
228 
229  return computedOdometry;
230 }
