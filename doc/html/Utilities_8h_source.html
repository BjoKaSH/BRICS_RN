60  inline double perpendicularDistance(const FrameWithId& point,
61  const FrameWithId& linePoint1, const FrameWithId& linePoint2) {
62 
63  double x0 = point.getFrame().p.x();
64  double y0 = point.getFrame().p.y();
65 
66  double x1 = linePoint1.getFrame().p.x();
67  double y1 = linePoint1.getFrame().p.y();
68 
69  double x2 = linePoint2.getFrame().p.x();
70  double y2 = linePoint2.getFrame().p.y();
71 
72  double dist = fabs((x2 - x1)*(y1 - y0)-(x1 - x0)*(y2 - y1)) /
73  sqrt((x2 - x1)*(x2 - x1)+(y2 - y1)*(y2 - y1));
74 
75  return dist;
76 
77  }

88  inline void prunePath(const std::vector<FrameWithId>& path,
89  const FrameWithId& point,

91 
92  std::vector<FrameWithId> localPath;

95  double dmax = perpendicularDistance(localPath[0], localPath[1], point);
96  unsigned int index = 1;
97 
98  for (unsigned int i = 1; i < localPath.size(); i++) {
99  double d = perpendicularDistance(localPath[i - 1], localPath[i],
100  point);
101  if (d <= dmax) {
102  index = i;
103  dmax = d;
104  }
105  }
106 
107  prunedPath.push_back(point);
108  prunedPath.insert(prunedPath.end(), localPath.begin() + index, localPath.end());
109 
110  }
