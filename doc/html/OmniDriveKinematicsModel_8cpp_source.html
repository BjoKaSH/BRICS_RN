54 void OmniDriveKinematicsModel::convert(const double& x1, const double& y1, const double& theta, double& x2, double& y2, bool inverse) {
55 
56  double k = (inverse == true) ? 1 : -1;
57 
58  x2 = (x1 * cos(theta) + y1 * sin(theta)) * k;
59  y2 = (y1 * cos(theta) - x1 * sin(theta)) * k;
60 }
